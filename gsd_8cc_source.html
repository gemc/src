63 G4bool GSensitiveDetector::ProcessHits(G4Step* thisStep, G4TouchableHistory* g4th)
64 {
65  // gDynamicDigitizationLocalInstance is guaranteed to exist in GSensitiveDetector::Initialize
66 
67 
68  double depe = thisStep->GetTotalEnergyDeposit();
69 
70  // if there is a decision to skip this event, return
71  if( decisionToSkipHit(depe) ) {
72  return true;
73  }
74 
75  // get the vector of GTouchables returned by gDynamicDigitizationLocalInstance
76  // if not defined by the plugin, base class will return a vector with one element, the input
77  vector<GTouchable*> thisStepProcessedTouchables = gDynamicDigitizationLocalInstance->processTouchable(getGTouchable(thisStep), thisStep);
78 
79  for(const auto thisGTouchable: thisStepProcessedTouchables) {
80 
81  if(isThisANewTouchable(thisGTouchable)) {
82  // new ghit, insert in gHitsCollection
83  // the constructor takes care of the filling the hit according to gHitBitSet
84  gHitsCollection->insert(new GHit(thisGTouchable, thisStep, gHitBitSet));
85 
86  } else {
87  // not a new touchable, must be in hit collection
88  // retrieve hit from hit collection
89  // add informations according to gHitBitSet
90  GHit* existingHit = getHitInHitCollectionUsingTouchable(thisGTouchable);
91  existingHit->addHitInfos(gHitBitSet, thisStep);
92  }
